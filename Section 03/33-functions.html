33 - Functions

A function allows you to reuse a piece of code. Also, receive data and return data back. Functions are useable for implementing DRY (Do not Repeat Yourself) code.

function logger() {
console.log("My name is Michael.");
}

// Calling / Running / Invoking the function
logger();

// Running a function that passes through data in the parameters.
function fruitProcessor(apples, oranges) {
const juice = `Juice with ${apples} apples and ${oranges} oranges.`;
return juice;
}

const yourDrink = fruitProcessor(5, 4);
console.log(yourDrink);

const yourOtherDrink = fruitProcessor(15, 25);
console.log(yourOtherDrink);

// Another example
function venadasFamily(husband, wife, child) {
console.log(husband, wife, child);
const family = `Meet the Venadas family, ${husband} , ${wife} , and ${child}!`;
return family;
}

const names = venadasFamily("Michael", "Rosie", "Colin");
console.log(names);
