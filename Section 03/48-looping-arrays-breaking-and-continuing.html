48 - Looping Arrays, Breaking, and Continuing

Looping data from an array

const michael = [
"Michael",
"Venadas",
2023 - 1979,
"Front End Developer",
["Richard", "Mauro", "Josh"],
true,
];

for (let i = 0; i < michael.length; i++) {
console.log(michael[i]);
}

Pusing data into an empty array

const years = [1979, 1984, 2018];
const ages = [];

for (i = 0; i < years.length; i++) {
ages.push(2023 - years[i]);
}

console.log(ages);

Continue
console.log("---- ONLY STRINGS ----");

for (i = 0; i < michael.length; i++) {
if (typeof michael[i] !== "string") continue;
console.log(michael[i], typeof michael[i]);
}

Break
console.log("---- BREAK WITH NUMBER ----");

for (i = 0; i < michael.length; i++) {
if (typeof michael[i] === "number") break;
console.log(michael[i], typeof michael[i]);
}
