const temperatures = [3, -2, -6, -1, 'error', 9, 13, 17, 15, 14, 9, 5];

const calcTempAmplitude = function (temps) {
let max = temps[0];
let min = temps[0];

for (let i = 0; i < temps.length; i++) {
const curTemp = temps[i];
if (typeof curTemp !== 'number') continue;
if (curTemp > max) max = curTemp;
if (curTemp < min) min = curTemp;
}

console.log(max, min);
};

calcTempAmplitude([3, 7, 4, 1, 8]);

My example

Array of lotto numbers, we want to find out today's winner. The winning lotto number consist of the highest average. Note: In this example I used MDN Web Docs to research and find the reduce method that helped me add up the numbers in the array.


Task #1 - Add up all the numbers for each lotto player

const michaelLotto = [123, 222, 848, 991, 10, 78];
const richardLotto = [113, 22, 818, 490, 2, 78];

let initialValue = 0;
let michaelTotal = michaelLotto.reduce(
(set1, set2) => set1 + set2,
initialValue
);

let richardTotal = richardLotto.reduce(
(set1, set2) => set1 + set2,
initialValue
);

console.log(michaelTotal, richardTotal);

Task #2 - Find the highest average of the 2 player
function whoWins() {
if (michaelTotal > richardTotal) {
console.log('Mike wins the lotto!');
} else {
console.log('Richard wins the lotto!');
}
}

const winner = whoWins();
console.log(winner);
